)

    basic_auth=shared.AuthServiceRequestBodyBasicAuth(
        password='GcDLHesCr6iCzfh',
        username='Celestine.Hayes',

            expected_value='channels',

        ),
    ],
)

    api_key_auth_new="Token <YOUR_API_KEY>",
    oauth2="Bearer YOUR_OAUTH2_TOKEN",
))

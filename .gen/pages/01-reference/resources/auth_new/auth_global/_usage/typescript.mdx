import { SDK } from "openapi";

(async() => {
  const sdk = new SDK({
    security: {
      apiKeyAuth: "Token YOUR_API_KEY",
    },
    globalPathParam: 100,
    globalQueryParam: "some example global query param",
  });

  const res = await sdk.authNew.authGlobal({
    basicAuth: {
      password: "mLRkaeXTziRHM2j",
      username: "Myra56",
    },
    headerAuth: [
      {
        expectedValue: "East",
        headerName: "Fadel thistle",
      },
    ],
  });

  if (res.statusCode == 200) {
    // handle response
  }
})();

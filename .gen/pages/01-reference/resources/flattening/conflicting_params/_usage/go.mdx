```go showLineNumbers copy filename="Conflicting Params usage example"
package main

import(
	"context"
	"log"
	"openapi"
	"openapi/pkg/models/shared"
)

func main() {
    s := openapi.New(
        openapi.WithSecurity(shared.Security{
            APIKeyAuth: openapi.String("Token YOUR_API_KEY"),
        }),
        openapi.WithGlobalPathParam(100),
        openapi.WithGlobalQueryParam("some example global query param"),
    )


    var strPathParameter string = "instead"

    var strQueryParameter string = "Customer"

    ctx := context.Background()
    res, err := s.Flattening.ConflictingParams(ctx, strPathParameter, strQueryParameter)
    if err != nil {
        log.Fatal(err)
    }

    if res.Res != nil {
        // handle response
    }
}
```

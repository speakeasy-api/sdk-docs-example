import { SDK } from "openapi";

(async() => {
  const sdk = new SDK({
    security: {
      apiKeyAuth: "Token YOUR_API_KEY",
    },
    globalPathParam: 100,
    globalQueryParam: "some example global query param",
  });

  const res = await sdk.requestBodies.requestBodyWriteOnly({
    bool: false,
    num: 4090.54,
    string: "gold",
  });

  if (res.statusCode == 200) {
    // handle response
  }
})();

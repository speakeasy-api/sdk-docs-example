    security: {
      apiKeyAuth: "Token YOUR_API_KEY",

    globalPathParam: 100,

  const res = await sdk.authNew.apiKeyAuthGlobalNew({
    basicAuth: {
      password: "qIEJuENlnSNVFxp",
      username: "Antonetta64",
    },
    headerAuth: [
      {
        expectedValue: "stable",
        headerName: "Bicycle wonderful TCP",
      },
    ],
  });

  if (res.statusCode == 200) {
    // handle response
  }
})();
